Dashboard => Manage Jenkins => Script Console

def jobName = "JOB_FOLDER_NAME"  
def job = Jenkins.instance.getItem(jobName)  
job.getBuilds().each { it.delete() }  
job.nextBuildNumber = 1   
job.save()
